198int websocket_read_text(SOCKET client, char *out, int out_size) {
199 unsigned char hdr[2];
200 int r = recv(client, (char *)hdr, 2, 0);
201 if (r != 2)
202 return -1;
203 unsigned char opcode = hdr[0] & 0x0F;
204 unsigned char masked = (hdr[1] & 0x80) ? 1 : 0;
205 size_t len = hdr[1] & 0x7F;
206 if (opcode == 0x8)
207 return 0; // close
208 // handle ping (0x9): read payload and reply with pong
209 if (opcode == 0x9) {
210 if (!masked)
211 return -1;
212 if (len == 126) {
213 unsigned char ext[2];
214 if (recv(client, (char *)ext, 2, 0) != 2)
215 return -1;
216 len = (ext[0] << 8) | ext[1];
217 } else if (len == 127) {
218 return -1;
219 }
220 unsigned char mask[4];
221 if (recv(client, (char *)mask, 4, 0) != 4)
222 return -1;
223 unsigned char *payload = (unsigned char *)malloc(len);
224 if (!payload)
225 return -1;
226 size_t got = 0;
227 while (got < len) {
228 int chunk = recv(client, (char *)payload + got, (int)(len - got), 0);
229 if (chunk <= 0) {
230 free(payload);
231 return -1;
232 }
233 got += chunk;
234 }
235 for (size_t i = 0; i < len; ++i)
236 payload[i] ^= mask[i % 4];
237 // send pong frame (server frames are not masked)
238 unsigned char frame_hdr[2];
239 frame_hdr[0] = 0x8A; // FIN + pong
240 if (len <= 125) {
241 frame_hdr[1] = (unsigned char)len;
242 send(client, (const char *)frame_hdr, 2, 0);
243 if (len)
244 send(client, (const char *)payload, (int)len, 0);
245 }
246 free(payload);
247 return -2; // indicate control frame handled
248 }
249 if (!masked)
250 return -1; // client-to-server must be masked
251 if (len == 126) {
252 unsigned char ext[2];
253 if (recv(client, (char *)ext, 2, 0) != 2)
254 return -1;

256 } else if (len == 127) {
257 return -1; // too big for demo
258 }

261 return -1;
262 if (len >= (size_t)out_size)
263 return -1;

265 if (!payload)

277 payload[i] ^= mask[i % 4];
278 }

283}
